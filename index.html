function noteIdx(n){
  return EN.indexOf(
    n
      .replace('Db','C#')
      .replace('Eb','D#')
      .replace('Gb','F#')
      .replace('Ab','G#')
      .replace('Bb','A#')
  );
}

function midiToName(m,lang='en'){
  const i = (m % 12 + 12) % 12;
  const o = Math.floor(m / 12) - 1;
  return (lang === 'es' ? ES[i] : EN[i]) + o;
}

function midi(note){
  const m = /^([A-G])(#|b)?(\d+)$/.exec(note);

  if(!m)
    return 60;

  let idx = noteIdx(m[1] + (m[2] || ''));

  return (+m[3] + 1) * 12 + idx;
}

function hz(m){
  return 440 * Math.pow(2,(m - 69) / 12);
}

function setupAudio(){
  if(!AC){
    const C = window.AudioContext || window.webkitAudioContext;
    AC = new C();

    master = AC.createGain();
    compressor = AC.createDynamicsCompressor();

    master.connect(compressor);
    compressor.connect(AC.destination);
  }

  if(AC.state === 'suspended')
    AC.resume();

  master.gain.value = +$('#volume').value || .55;
}

function env(preset){
  const r = +$('#release').value || .45;

  return {
    piano:{a:.01,d:.18,s:.35,r},
    organ:{a:.02,d:.05,s:.85,r:.18},
    synth:{a:.03,d:.12,s:.5,r},
    bass:{a:.005,d:.15,s:.42,r:.28},
    bell:{a:.005,d:.45,s:.18,r:1.2}
  }[preset];
}

function waveForPreset(){
  const p = $('#preset').value;

  return p === 'piano'
    ? 'triangle'
    : p === 'organ'
      ? 'sine'
      : p === 'bass'
        ? 'sawtooth'
        : p === 'bell'
          ? 'sine'
          : $('#wave').value;
}

function noteOn(m,velocity=.8,source='ui'){
  setupAudio();

  m += +$('#transpose').value || 0;

  const id = source + ':' + m;

  if(active.has(id))
    noteOff(id);

  const e = env($('#preset').value);
  const now = AC.currentTime;

  const o = AC.createOscillator();
  const g = AC.createGain();
  const filter = AC.createBiquadFilter();

  o.type = waveForPreset();
  o.frequency.value = hz(m);

  filter.type = 'lowpass';
  filter.frequency.value =
    $('#preset').value === 'bass'
      ? 900
      : $('#preset').value === 'bell'
        ? 4200
        : 2800;

  g.gain.setValueAtTime(0,now);
  g.gain.linearRampToValueAtTime(velocity,now + e.a);
  g.gain.linearRampToValueAtTime(velocity * e.s,now + e.a + e.d);

  o.connect(filter);
  filter.connect(g);
  g.connect(master);
  o.start(now);

  active.set(id,{
    o,
    g,
    e,
    base:m
  });

  markKey(m,true);

  app.notes++;

  if(rec){
    rec.events.push({
      t:performance.now() - rec.start,
      type:'on',
      m
    });
  }

  if(quiz && quiz.active)
    quizAnswer(m);

  stats();

  return id;
}

function noteOff(id){
  if(!AC)
    return;

  const n = active.get(id);

  if(!n)
    return;

  const now = AC.currentTime;

  n.g.gain.cancelScheduledValues(now);
  n.g.gain.setValueAtTime(Math.max(n.g.gain.value,.0001),now);
  n.g.gain.exponentialRampToValueAtTime(.0001,now + n.e.r);

  n.o.stop(now + n.e.r + .03);

  active.delete(id);
  markKey(n.base,false);

  if(rec){
    rec.events.push({
      t:performance.now() - rec.start,
      type:'off',
      m:n.base
    });
  }
}

function isWhiteMidi(m){
  return WHITE.includes((m % 12 + 12) % 12);
}

function buildPiano(){
  const total = +$('#rangeSel').value || 49;
  const start = +$('#startOct').value || 2;
  const startMidi = (start + 1) * 12;

  const piano = $('#piano');
  piano.innerHTML = '';

  const whiteW =
    total >= 88 ? 40 :
    total >= 61 ? 48 :
    56;

  const blackW = Math.round(whiteW * 0.62);
  const gap = 2;

  let whiteCount = 0;
  const whitePositions = new Map();
  let keyIndex = 0;

  for(let i = 0; i < total; i++){
    const m = startMidi + i;

    if(isWhiteMidi(m)){
      const left = whiteCount * (whiteW + gap);
      whitePositions.set(m,left);

      const k = keyEl(
        m,
        'white',
        left,
        whiteW,
        keyIndex
      );

      piano.append(k);

      whiteCount++;
      keyIndex++;
    }
  }

  for(let i = 0; i < total; i++){
    const m = startMidi + i;

    if(!isWhiteMidi(m)){
      const prevWhite = findPreviousWhite(m,startMidi);
      const prevLeft = whitePositions.get(prevWhite);

      if(prevLeft === undefined)
        continue;

      const left = prevLeft + whiteW - (blackW / 2);

      const k = keyEl(
        m,
        'black',
        left,
        blackW,
        keyIndex
      );

      piano.append(k);

      keyIndex++;
    }
  }

  piano.style.minWidth = (whiteCount * (whiteW + gap) + 40) + 'px';

  markTheory(false);
}

function findPreviousWhite(m,startMidi){
  for(let x = m - 1; x >= startMidi; x--){
    if(isWhiteMidi(x))
      return x;
  }

  return startMidi;
}

function keyEl(m,cls,left,w,index){
  const b = document.createElement('button');

  b.className = 'key ' + cls;
  b.style.left = left + 'px';
  b.style.width = w + 'px';
  b.dataset.midi = m;
  b.dataset.keyIndex = index;
  b.setAttribute('aria-label',midiToName(m,'en'));

  const sp = document.createElement('span');
  sp.innerHTML = labelFor(m,index);
  b.append(sp);

  let id = null;

  b.addEventListener('pointerdown',e => {
    e.preventDefault();
    b.setPointerCapture?.(e.pointerId);
    id = noteOn(m,.7,'ptr');
  });

  b.addEventListener('pointerup',() => {
    if(!$('#sustain').checked && id)
      noteOff(id);

    id = null;
  });

  b.addEventListener('pointerleave',() => {
    if(!$('#sustain').checked && id)
      noteOff(id);

    id = null;
  });

  return b;
}

function markTheory(output=true){
  $$('.key').forEach(k => {
    k.classList.remove('inScale','root','target');
  });

  const notes = currentFormula();
  const rootPc = notes[0] % 12;

  $$('.key').forEach(k => {

        pc === rootPc ? 'root' : 'inScale'

      .map(m => midiToName(m,'es'))
      .join(' · ');

      `${$('#rootSel').selectedOptions[0].text} ${$('#formulaSel').value}\n${names}`;

function renderSong(){
  const s = SONGS[+$('#songSel').value || 0];
  const tr = +$('#songTranspose').value || 0;

  const tokens = s.notes.map(n => {
    return midiToName(
      midi(n) + tr,
      $('#labels').value === 'es' ? 'es' : 'en'
    );
  });

  let lines = [];

  for(let i = 0; i < tokens.length; i += 8){
    lines.push(tokens.slice(i,i + 8).join('  '));
  }

  $('#songInfo').textContent =
    `BPM ${s.bpm} · ${s.notes.length} notas`;

  $('#songNotes').textContent =
    lines.join('\n');
}

function playSong(){
  if(songTimer)
    return;

  const s = SONGS[+$('#songSel').value || 0];
  const speed = +$('#songSpeed').value || 1;
  const tr = +$('#songTranspose').value || 0;
  const gap = 60000 / (s.bpm * speed);

  songIndex = 0;
  renderSong();

  songTimer = setInterval(() => {
    if(songIndex >= s.notes.length){
      stopSong();
      return;
    }

    const m = midi(s.notes[songIndex]) + tr;
    const id = noteOn(m,.6,'song');

    setTimeout(() => {
      noteOff(id);
    },gap * .72);

    songIndex++;
  },gap);
}

function stopSong(){
  clearInterval(songTimer);
  songTimer = null;
  songIndex = 0;
  panic();
}

function clickSound(time,accent=false){
  setupAudio();

  const vol = +$('#clickVol').value || .5;
  const o = AC.createOscillator();
  const g = AC.createGain();

  o.type = 'square';
  o.frequency.value = accent ? 1450 : 900;

  g.gain.setValueAtTime(.0001,time);
  g.gain.exponentialRampToValueAtTime(
    vol * (accent ? .55 : .32),
    time + .005
  );
  g.gain.exponentialRampToValueAtTime(
    .0001,
    time + .07
  );

  o.connect(g);
  g.connect(master);

  o.start(time);
  o.stop(time + .08);
}

function metroSchedule(){
  const bpm = +$('#bpm').value || 90;
  const sub = +$('#sub').value || 1;
  const meter = +$('#meter').value || 4;
  const step = 60 / bpm / sub;

  while(nextClick < AC.currentTime + .12){
    const b = Math.floor(beat / sub);
    const s = beat % sub;

    clickSound(
      nextClick,
      b === 0 && s === 0
    );

    setTimeout(() => {
      $$('.beat').forEach((x,i) => {
        x.classList.toggle('on',i === b);
      });
    },Math.max(0,(nextClick - AC.currentTime) * 1000));

    nextClick += step;
    beat = (beat + 1) % (meter * sub);
  }
}

function tap(){
  const now = performance.now();

  taps = taps.filter(t => now - t < 2500);
  taps.push(now);

  if(taps.length >= 4){
    const dif = taps
      .slice(1)
      .map((t,i) => t - taps[i]);

    const avg =
      dif.reduce((a,b) => a + b,0) / dif.length;

    $('#bpm').value =
      Math.round(60000 / avg);

    toast('BPM: ' + $('#bpm').value);
  }
}

    right:0,
    target:null
  };

  nextQuiz();
}

function nextQuiz(){
  if(!quiz || quiz.done >= quiz.total)
    return finishQuiz();

  const keys = $$('.key');

  const pool =
    $('#quizRange').value === 'middle'
      ? keys.filter(k => +k.dataset.midi >= 60 && +k.dataset.midi <= 72)
      : keys;

  quiz.target =
    +pool[Math.floor(Math.random() * pool.length)].dataset.midi;

  const mode = $('#quizMode').value;

  if(mode === 'ear'){
    const id = noteOn(quiz.target,.55,'quiz');

    setTimeout(() => {
      noteOff(id);
    },700);

    $('#quizBox').innerHTML =
      `Pregunta ${quiz.done + 1}/${quiz.total}: escucha la nota y tócala en el teclado.`;
  }
  else if(mode === 'chord'){
    const root = Math.floor(Math.random() * 12);
    const names = Object.keys(CHORDS);
    const ch = names[Math.floor(Math.random() * names.length)];

    const notes =
      CHORDS[ch].map(x => 60 + root + x);

    quiz.chord = {
      root,
      ch,
      notes
    };

    notes.forEach((m,i) => {
      setTimeout(() => {
        const id = noteOn(m,.45,'quizch');

        setTimeout(() => {
          noteOff(id);
        },700);
      },i * 60);
    });

    $('#quizBox').innerHTML =
      `Pregunta ${quiz.done + 1}/${quiz.total}: toca la tónica del acorde que escuchaste.`;
  }
  else{
    $('#quizBox').innerHTML =
      `Pregunta ${quiz.done + 1}/${quiz.total}: toca ${midiToName(quiz.target,'es')}.`;
  }
}

function quizAnswer(m){
  if(!quiz || !quiz.active)
    return;

  let ok = false;

  if($('#quizMode').value === 'chord'){
    ok = (m % 12) === quiz.chord.root;
  }else{
    ok = (m % 12) === (quiz.target % 12);
  }

  quiz.done++;

  if(ok){
    quiz.right++;
    app.quiz.right++;
  }

  app.quiz.total++;
  save();

  $('#quizBox').innerHTML =
    (ok ? '✅ Correcto' : '❌ Revisa') +
    ` · ${quiz.right}/${quiz.done}`;

  setTimeout(nextQuiz,700);
}

function finishQuiz(){
  if(!quiz)
    return;

  $('#quizBox').innerHTML =
    `Resultado: ${quiz.right}/${quiz.total}`;

  quiz.active = false;
  quiz = null;

  stats();
}

async function enableMIDI(){
  if(!navigator.requestMIDIAccess){
    toast('Web MIDI no está disponible en este navegador');
    return;
  }

  try{
    const access = await navigator.requestMIDIAccess();

    for(const input of access.inputs.values()){
      input.onmidimessage = e => {
        const [cmd,note,vel] = e.data;

        if((cmd & 240) === 144 && vel > 0){
          noteOn(note,vel / 127,'midi');
        }

        if((cmd & 240) === 128 || ((cmd & 240) === 144 && vel === 0)){
          [...active.keys()]
            .filter(k => k.endsWith(':' + note))
            .forEach(noteOff);
        }
      };
    }

    toast('MIDI activado');
  }catch{
    toast('No se pudo activar MIDI');
  }
}

function stats(){
  $('#statNotes').textContent = app.notes || 0;
  $('#statRec').textContent = app.recs.length;
  $('#statSongs').textContent = SONGS.length;

  const q = app.quiz || {
    right:0,
    total:0
  };

  $('#statQuiz').textContent =
    q.total
      ? Math.round(q.right / q.total * 100) + '%'
      : '0%';
}

function go(id){
  $$('.screen').forEach(s => {
    s.classList.toggle('active',s.id === id);
  });

  $$('.tab').forEach(t => {
    t.classList.toggle('active',t.dataset.screen === id);
  });

  window.scrollTo({
    top:0,
    behavior:'smooth'
  });
}

function bind(){
  document.documentElement.dataset.theme =
    app.theme || 'dark';

  initSelects();
  buildPiano();
  setupDots();
  renderRecs();
  stats();

  $$('.tab').forEach(b => {
    b.onclick = () => go(b.dataset.screen);
  });

  $$('[data-go]').forEach(b => {
    b.onclick = () => go(b.dataset.go);
  });

  $('#themeBtn').onclick = () => {
    app.theme = app.theme === 'dark' ? 'light' : 'dark';
    document.documentElement.dataset.theme = app.theme;
    save();
  };

  $('#printBtn').onclick = () => print();

  [
    'rangeSel',
    'startOct',
    'labels',
    'showComputerKeys'
  ].forEach(id => {
    $('#' + id).onchange = buildPiano;
  });

  [
    'preset',
    'volume',
    'release',
    'wave'
  ].forEach(id => {
    $('#' + id).onchange = () => {};
  });

  $('#panicBtn').onclick = panic;
  $('#midiBtn').onclick = enableMIDI;

  $('#theoryType').onchange = () => {
    updateFormulaOptions();
    markTheory(true);
  };

  [
    'rootSel',
    'formulaSel',
    'theoryOct'
  ].forEach(id => {
    $('#' + id).onchange = () => markTheory(true);
  });

  $('#showTheory').onclick = () => markTheory(true);
  $('#playTheory').onclick = playTheory;

  $('#clearMarks').onclick = () => {
    $$('.key').forEach(k => {
      k.classList.remove('inScale','root','target');
    });
  };

  [
    'songSel',
    'songTranspose',
    'songSpeed'
  ].forEach(id => {
    $('#' + id).onchange = renderSong;
  });

  $('#playSong').onclick = playSong;
  $('#stopSong').onclick = stopSong;
  $('#practiceSong').onclick = practiceSong;

  $('#recStart').onclick = startRec;
  $('#recStop').onclick = stopRec;
  $('#recPlay').onclick = () => playRec();
  $('#recSave').onclick = saveRec;

  $('#metroStart').onclick = startMetro;
  $('#metroStop').onclick = stopMetro;
  $('#meter').onchange = setupDots;
  $('#tapTempo').onclick = tap;

  $('#bpmUp').onclick = () => {
    $('#bpm').value = (+$('#bpm').value || 90) + 5;
  };

  $('#bpmDown').onclick = () => {
    $('#bpm').value = Math.max(30,(+$('#bpm').value || 90) - 5);
  };

  $('#quizStart').onclick = startQuiz;
  $('#quizStop').onclick = finishQuiz;

  document.addEventListener('click',e => {
    if(e.target.dataset.recPlay){
      playRec(app.recs[+e.target.dataset.recPlay].events);
    }

    if(e.target.dataset.recDel){
      app.recs.splice(+e.target.dataset.recDel,1);
      save();
      renderRecs();
    }
  });

  document.addEventListener('keydown',e => {
    if(['INPUT','TEXTAREA','SELECT'].includes(document.activeElement.tagName))
      return;

    const idx = KEYMAP.indexOf(e.key.toLowerCase());

    if(idx >= 0 && !e.repeat){
      const k = document.querySelector(`[data-key-index="${idx}"]`);

      if(k)
        noteOn(+k.dataset.midi,.65,'kb' + idx);
    }

    if(e.code === 'Space'){
      e.preventDefault();

      metro
        ? stopMetro()
        : startMetro();
    }

    if(e.key.toLowerCase() === 't')
      go('teclado');

    if(e.key.toLowerCase() === 'm')
      go('metronomo');
  });

  document.addEventListener('keyup',e => {
    const idx = KEYMAP.indexOf(e.key.toLowerCase());

    if(idx >= 0 && !$('#sustain').checked){
      [...active.keys()]
        .filter(k => k.startsWith('kb' + idx + ':'))
        .forEach(noteOff);
    }
  });

  window.addEventListener('beforeinstallprompt',e => {
    e.preventDefault();
    deferredPrompt = e;
    $('#installBtn').hidden = false;
  });

  $('#installBtn').onclick = async () => {
    if(deferredPrompt){
      deferredPrompt.prompt();
      deferredPrompt = null;
      $('#installBtn').hidden = true;
    }
  };

  if('serviceWorker' in navigator){
    window.addEventListener('load',() => {
      navigator.serviceWorker
        .register('./sw.js')
        .catch(() => {});
    });
  }
}

bind();
